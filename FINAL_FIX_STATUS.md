### 核心策略：100% 复用现有功能
```javascript
// 极简实现 - 只有 ~80 行代码
class NewTabPage {
    constructor() {
        this.modalManager = null;  // 直接使用现有的 ModalManager
        this.pinnedLinks = [];     // 只管理固定链接
        this.init();
    }
    
    async setup() {
        // 1. 加载固定链接
        await this.loadPinnedLinks();
        
        // 2. 使用现有 ModalManager (包含搜索、主题、键盘导航等所有功能)
        this.modalManager = new ModalManager();
        
        // 3. 最小适配：只修改书签打开方式
        this.adaptForNewTab();
        
        // 4. 固定链接管理
        this.bindPinnedLinksEvents();
        this.renderPinnedLinks();
    }
}
```
